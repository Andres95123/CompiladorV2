INIT ::= SENTENCES EOF { System.out.println("Fin del compilado"); } ;
SENTENCES ::= ASSIGNACION | IF_SENTENCE ;
ASSIGNACION ::= ID ASSIGN OPERACIONES EOL ;
OPERACIONES ::= OPERACIONES + OPERACIONES | OPERACIONES - OPERACIONES | OPERACIONES * OPERACIONES | OPERACIONES / OPERACIONES | OPERACIONES % OPERACIONES | NUM ;
IF_SENTENCE ::= IF OPEN_BRACKET IF_CONTENT EOL ;
IF_CONTENT ::= SENTENCES IF_CONTENT | SENTENCES CLOSE_BRACKET ;
